CREATE TABLE user(
    id INTEGER PRIMARY KEY NOT NULL,
    username VARCHAR(50),
    password LONGVARCHAR,
    type INTEGER
);

CREATE TABLE produit(
    id INTEGER PRIMARY KEY NOT NULL,
    prix REAL,
    type INTEGER,
    adresse VARCHAR(100),
    nb_places INTEGER,
    IDloueur INTEGER NOT NULL,
    FOREIGN KEY(IDloueur) REFERENCES user(id)
);

DROP TABLE produit;
DROP TABLE user;
